#!/bin/bash
#run as root
rpm -Uvh https://repo.zabbix.com/zabbix/4.4/rhel/7/x86_64/zabbix-release-4.4-1.el7
yum -y install zabbix-agent
chkconfig zabbix-agent on
sed -i 's/Server=127.0.0.1/Server=192.168.56.103/' /etc/zabbix/zabbix_agentd.conf
sed -i 's/ServerActive=127.0.0.1/ServerActive=192.168.56.103/' /etc/zabbix/zabbix_agentd.conf
sed -i "s/Hostname=Zabbix\ server/Hostname=zabbix agent/" /etc/zabbix/zabbix_agentd.conf
  service zabbix-agent restart
